fun M_rxSubscriber(): Unit

Implement rx subscriber wrapping coroutine block

        val flowable = Flowable
                .interval(300, TimeUnit.MILLISECONDS)
                .take(10)
        val subscriber = RxSubscriber<Long> {
            receive().p
//            close()
            receive().p
            receive().p
//            delay(1500)
            receive().p
            receive().p
            receive().p
        }
        flowable.subscribe(subscriber)
        Thread.sleep(10000)
