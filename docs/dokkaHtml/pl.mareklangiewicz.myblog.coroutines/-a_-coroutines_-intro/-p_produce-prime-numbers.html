fun P_producePrimeNumbers(): Unit

Generate prime numbers with crazy channels pipeline

sample {
        var cur = produceNumbersFrom(CommonPool, 2)
        for (i in 1..20) {
            val prime = cur.receive()
            "prime: $prime".p
            cur = cur.filter(context) { it % prime != 0 }
        }
    }
