fun N_fibonacciChannels(): Unit

Use channels to implement fibonacci numbers generator

val c = Channel<Int>(2) // experiment with different capacity
launch(CommonPool) {
    fibonacci(10, c)
}
launch(CommonPool) {
    for (x in c) {
        delay(300)
        "*********************** received $x".p
    }
}
